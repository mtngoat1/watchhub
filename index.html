                async function loadMedia(endpoint, headingText) {

                                if (!response.ok) throw new Error('API down or waking up. Free accounts take 1 minute to wake up.');

                                const data = await response.json();

                                const items = Array.isArray(data) ? data : (data.results || data.media || []);

                                if (items.length === 0) { mediaGrid.innerHTML = '<div class="col-span-full text-center text-gray-500 py-12 text-lg">No media found.</div>'; return; }

                                items.forEach(item => {

                                        const title = item.title || item.name || 'Unknown Title';

                                        const image = item.image || item.poster || 'https://placeholder.com';

                                        const link = item.link || item.url || '';

                                        const card = document.createElement('div');

                                        card.className = 'bg-gray-900 rounded-lg overflow-hidden border border-gray-800 hover:border-red-500 transition duration-300 cursor-pointer flex flex-col group';

                                        card.addEventListener('click', () => playMedia(title, link));

                                });

                function playMedia(title, streamUrl) {

                        if (!streamUrl) { alert('No video source found.'); return; }

                        playerTitle.textContent = `Now Playing: ${title}`;

                        playerSection.classList.remove('hidden');

                        playerSection.scrollIntoView({ behavior: 'smooth' });

                }


                function closePlayer() { playerSection.classList.add('hidden'); videoIframe.src = ''; }
